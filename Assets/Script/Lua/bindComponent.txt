import "UnityEngine"
if not UnityEngine.GameObject then
    error("Click Make/All to generate lua wrap file")
end

local  class ={}

function main()
	print("Run Binding Component")
	class.main = GameObject.Find("Main"):GetComponent("KnifeType")
	return class
end

function setUIInit()
	class.resultText = class.main:getBindingComponent("ResultText","Text")
end

function setResultText()
	class.resultText.text = "Set Result Value"
end
