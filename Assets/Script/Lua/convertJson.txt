function getJson(jsonValue)
    return jsonToTable(jsonValue)
end

function jsonToTable(str)
    local result = {}
    if stringIsEmpty(str) then
        return result
    end
    str = string.sub(str, 2, #str - 1)
    local jsonSplite = string.split(str, ',')
    for _, s in ipairs(jsonSplite) do
        local keyValue = string.split(s, ':')
        local key = string.sub(keyValue[1], 2, #keyValue[1] - 1)
        local value = keyValue[2]
        if string.match(value, "false") ~= nil then
            value = false
        elseif string.match(value, "true") ~= nil then
            value = true
        elseif tonumber(value) ~= nil then
            value = tonumber(value)
        end
        result[key] = value
    end
    return result
end

function stringIsEmpty(value)
    if type(value) ~= "string" then
        return true
    else
        return value == nil or value == ''
    end
end

string.split = function(s, p)

    local rt = {}
    string.gsub(s, '[^' .. p .. ']+', function(w)
        table.insert(rt, w)
    end)
    return rt

end
